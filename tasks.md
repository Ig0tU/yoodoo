     * @param {Element} section - The section container.

     * @param {number} index - 0-based index among elements of that type (default 0).

    static findElementInSection(section, type, index = 0) {
        // Map common names to the text shown in the element's badge (e.g., 'Headline')
        const typeMap = {
            headline: 'Headline',
            text: 'Text',

        };
        const displayName = typeMap[type.toLowerCase()] || type;
        const elements = Array.from(section.querySelectorAll('.yo-builder-element'));
        const filtered = elements.filter(el => {
            const label = el.querySelector('.uk-text-truncate')?.textContent.trim();
            return label && label.toLowerCase() === displayName.toLowerCase();
        });
        return filtered[index] || null;

     * Open the editor for a given element by clicking it.
     * Assumes the element is clickable (the whole element has a click handler).

        const clickable = element.querySelector('a[aria-label="Edit"]') || element;

        if (editor.tagName === 'TEXTAREA' || editor.tagName === 'INPUT') {
            editor.value = newText;
            editor.dispatchEvent(new Event('input', { bubbles: true }));
        } else if (editor.isContentEditable) {
            editor.textContent = newText;
            editor.dispatchEvent(new Event('input', { bubbles: true }));
        } else {
            throw new Error('Unsupported editor type');
        }
    }

     * Modify text at a specific character position.
     * @param {Element} editor - The editor element.
     * @param {number} position - 0-based character index where to start.
     * @param {string} replacement - The string to insert (can be empty to delete).
     * @param {number} deleteCount - Number of characters to delete after position.
     */
    static modifyEditorText(editor, position, replacement, deleteCount = 0) {
        let currentText = '';
        if (editor.tagName === 'TEXTAREA' || editor.tagName === 'INPUT') {
            currentText = editor.value;
        } else if (editor.isContentEditable) {
            currentText = editor.textContent;
        }

        const before = currentText.substring(0, position);
        const after = currentText.substring(position + deleteCount);
        const newText = before + replacement + after;

        if (editor.tagName === 'TEXTAREA' || editor.tagName === 'INPUT') {
            editor.value = newText;
            editor.dispatchEvent(new Event('input', { bubbles: true }));
            // Optionally set selection to the end of the inserted text
            const newPosition = before.length + replacement.length;
            editor.setSelectionRange(newPosition, newPosition);
        } else if (editor.isContentEditable) {

            editor.dispatchEvent(new Event('input', { bubbles: true }));
        }

static resolvePath(description) {
    // Simple pattern matching; can be expanded.
    // description: e.g., "section 'Team' > column 2 > headline 1"
    const tokens = description.split(/\s*>\s*/);
    let currentNode = this.layoutTree;
    for (const token of tokens) {
        const match = token.match(/^(section|row|column|headline|text|image|button)\s*(?:['"]?([^'"]+)['"]?)?(?:\s*(\d+))?$/i);
        if (!match) return null;
        const type = match[1].toLowerCase();
        const nameOrIndex = match[2] || match[3]; // either quoted name or index number
        const index = parseInt(match[3]) || 1;
        if (type === 'section' && nameOrIndex) {
            // find by name
            const section = currentNode.children.find(child => child.name && child.name.toLowerCase() === nameOrIndex.toLowerCase());
            if (!section) return null;
            currentNode = section;
        } else if (type === 'row') {
            // assume rows are ordered, index given
            const row = currentNode.children?.[index - 1];
            if (!row || row.type !== 'row') return null;
            currentNode = row;
        } else if (type === 'column') {
            const col = currentNode.children?.[index - 1];
            if (!col || col.type !== 'column') return null;
            currentNode = col;
        } else {
            // element type
            const elements = currentNode.children?.filter(child => child.type === type);
            const el = elements?.[index - 1];
            if (!el) return null;
            currentNode = el;
        }
    }
    return currentNode;
}
